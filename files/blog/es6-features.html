2. Arrow Functions

Shorter syntax cho functions v√† lexical this binding. 

// Traditional function
function add(a, b) {
    return a + b;
}

// Arrow function
const add = (a, b) => a + b;

// With single parameter

// With no parameters
const greet = () => console.log("Hello!");

// With block body
const calculate = (a, b) => {
    const sum = a + b;
    const product = a * b;
    return {sum, product};
};

// Lexical this
class Counter {
    constructor() {
        this.count = 0;
    }
    
    increment() {
        // Arrow function preserves this
        setInterval(() => {
            this.count++;
            console.log(this.count);
        }, 1000);
    }
}

8. Classes

10. Async/Await

// Async function
async function fetchUser(id) {
    try {
        const response = await fetch(`/api/users/${id}`);
        const user = await response.json();
        return user;
    } catch (error) {
        console.error("Error:", error);
        throw error;
    }
}

// Using async function
fetchUser(1)
    .then(user => console.log(user))
    .catch(error => console.error(error));

// Multiple awaits
async function getData() {
    const [users, posts] = await Promise.all([
        fetch('/api/users').then(r => r.json()),
        fetch('/api/posts').then(r => r.json())
    ]);
    return {users, posts};
}

// Async arrow function
const loadData = async () => {
    const data = await fetchData();
    return data;
};

14. Nullish Coalescing (ES2020)

// OR operator returns first truthy value
const value1 = 0 || "default"; // "default"
const value2 = "" || "default"; // "default"

// Nullish coalescing returns first defined value (not null/undefined)
const value3 = 0 ?? "default"; // 0
const value4 = "" ?? "default"; // ""
const value5 = null ?? "default"; // "default"
const value6 = undefined ?? "default"; // "default"

// Practical use
function greet(name) {
    const displayName = name ?? "Guest";
    return `Hello, ${displayName}!`;
}
